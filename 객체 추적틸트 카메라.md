## YOLOv8에서 특정 객체(예: 사람)만 필터링하기

OLOv8의 기본 모델(`yolov8n.pt`)은 사람, 자동차, 고양이 등 80가지의 사물을 한꺼번에 다 인식합니다. 짐벌 카메라가 오직 "사람(Person)"만 타겟팅하여 추적하게 만들려면, 추론 코드에서 다른 사물들은 전부 무시하고 `class id = 0` (사람) 데이터만 골라내도록 필터링을 걸어주어야 합니다.

#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from sensor_msgs.msg import Image
from cv_bridge import CvBridge
from ultralytics import YOLO
import cv2
import numpy as np

class Yolov8Node(Node):
    def __init__(self):
        super().__init__('yolov8_node')

        self.bridge = CvBridge()

        self.get_logger().info('YOLOv8 모델을 불러오는 중...')
        self.model = YOLO('yolov8n.pt')
        self.get_logger().info('YOLOv8 모델 로드 완료!')

        # 1. 원본 이미지 구독 (Subscribe)
        self.subscription = self.create_subscription(
            Image,
            '/image_raw', # 원본 카메라 토픽
            self.image_callback,
            10
        )

        # 2. 결과 이미지를 발행할 퍼블리셔(Publisher) 생성
        self.publisher_ = self.create_publisher(
            Image,
            '/yolo/result_image', # RViz에서 볼 새로운 토픽 이름
            10
        )

    def image_callback(self, msg):
        try:
            # yuv422_yuy2 수동 변환 유지
            if msg.encoding in ['yuv422_yuy2', 'yuyv']:
                img_data = np.frombuffer(msg.data, dtype=np.uint8).reshape(msg.height, msg.width, 2)
                cv_image = cv2.cvtColor(img_data, cv2.COLOR_YUV2BGR_YUYV)
            else:
                cv_image = self.bridge.imgmsg_to_cv2(msg, desired_encoding='bgr8')
        except Exception as e:
            self.get_logger().error(f'이미지 변환 에러: {e}')
            return

        if cv_image is None or cv_image.size == 0:
            return

        try:
            # YOLOv8 추론 실행
            results = self.model(cv_image, verbose=False)

            # 결과를 직접 그리기 위해 원본 이미지를 복사합니다.
            annotated_frame = cv_image.copy()

            # -------------------------------------------------------------
            # [★핵심 수정] 80가지 사물 중 오직 '사람(id=0)'만 필터링하여 그리기
            # -------------------------------------------------------------
            if len(results) > 0:
                boxes = results[0].boxes
                for box in boxes:
                    cls_id = int(box.cls[0])

                    # 0번(사람)이 아니라면 사각형을 그리지 않고 무시(패스)합니다!
                    if cls_id != 0:
                        continue

                    # 사람의 좌표 획득 (xyxy 형태)
                    x1, y1, x2, y2 = map(int, box.xyxy[0])
                    conf = float(box.conf[0]) # 인식 확률(신뢰도)

                    # 화면에 오직 사람에게만 초록색 사각형과 텍스트를 직접 그립니다.
                    cv2.rectangle(annotated_frame, (x1, y1), (x2, y2), (0, 255, 0), 2)
                    cv2.putText(annotated_frame, f"Person: {conf:.2f}", (x1, y1 - 10),
                                cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)

            # 3. 가공된 OpenCV 이미지를 다시 ROS 2 Image 메시지로 변환
            result_msg = self.bridge.cv2_to_imgmsg(annotated_frame, encoding="bgr8")

            # 4. RViz에서 시간 동기화를 위해 원본 이미지의 헤더 복사
            result_msg.header = msg.header

            # 토픽 발행!
            self.publisher_.publish(result_msg)

        except Exception as e:
            self.get_logger().error(f'YOLO 추론 중 에러 발생: {e}')


def main(args=None):
    rclpy.init(args=args)
    yolov8_node = Yolov8Node()

    try:
        rclpy.spin(yolov8_node)
    except KeyboardInterrupt:
        yolov8_node.get_logger().info('사용자에 의해 노드가 종료되었습니다.')
    finally:
        yolov8_node.destroy_node()
        rclpy.shutdown()

if __name__ == '__main__':
    main()

### 💻 2축 서보모터 자동 추적 통합 파이썬 코드

```
#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from sensor_msgs.msg import Image
from cv_bridge import CvBridge
from ultralytics import YOLO
import cv2
import numpy as np
import serial  # 아두이노 우노와의 시리얼 통신을 위한 라이브러리

class Yolov8GimbalNode(Node):
    def __init__(self):
        super().__init__('yolov8_gimbal_node')

        self.bridge = CvBridge()

        # -----------------------------------------------------------------
        # 1. 아두이노 우노 시리얼 포트 설정
        # -----------------------------------------------------------------
        self.arduino_port = '/dev/ttyACM0'  # 우노가 연결되는 표준 리눅스 포트
        self.baudrate = 115200              # 아두이노 스케치와 맞출 통신 속도

        try:
            self.serial_conn = serial.Serial(self.arduino_port, self.baudrate, timeout=0.1)
            self.get_logger().info(f'Success: Connected to Arduino Uno on {self.arduino_port}')
        except Exception as e:
            self.get_logger().error(f'Arduino Uno 연결 실패 (포트/권한 확인 필요): {e}')
            self.serial_conn = None

        # 짐벌 모터 초기 각도 설정 (정중앙 90도 배치)
        self.pan_angle = 90
        self.tilt_angle = 90

        # -----------------------------------------------------------------
        # 2. YOLOv8 모델 로드
        # -----------------------------------------------------------------
        self.get_logger().info('YOLOv8 모델을 불러오는 중...')
        self.model = YOLO('yolov8n.pt')
        self.get_logger().info('YOLOv8 모델 로드 완료!')

        # 3. 원본 이미지 구독 (Subscribe)
        self.subscription = self.create_subscription(
            Image,
            '/image_raw', 
            self.image_callback,
            10
        )

        # 4. 결과 이미지를 발행할 퍼블리셔(Publisher) 생성
        self.publisher_ = self.create_publisher(
            Image,
            '/yolo/result_image', 
            10
        )

    def image_callback(self, msg):
        try:
            # yuv422_yuy2 수동 변환 조건 유지
            if msg.encoding in ['yuv422_yuy2', 'yuyv']:
                img_data = np.frombuffer(msg.data, dtype=np.uint8).reshape(msg.height, msg.width, 2)
                cv_image = cv2.cvtColor(img_data, cv2.COLOR_YUV2BGR_YUYV)
            else:
                cv_image = self.bridge.imgmsg_to_cv2(msg, desired_encoding='bgr8')
        except Exception as e:
            self.get_logger().error(f'이미지 변환 에러: {e}')
            return

        if cv_image is None or cv_image.size == 0:
            return

        # 화면의 제어 기준점(정중앙 좌표) 계산 (예: 640x480 영상이면 320, 240)
        img_h, img_w, _ = cv_image.shape
        center_x = img_w // 2
        center_y = img_h // 2

        target_found = False
        error_x = 0
        error_y = 0

        try:
            # YOLOv8 추론 실행
            results = self.model(cv_image, verbose=False)
            annotated_frame = cv_image.copy()

            if len(results) > 0:
                boxes = results[0].boxes
                for box in boxes:
                    # -------------------------------------------------------------
                    # [앞서 했던 작업] 특정 객체 필터링 (0번 = 'person' 사람)
                    # -------------------------------------------------------------
                    cls_id = int(box.cls[0])
                    if cls_id != 0:
                        continue  # 사람이 아니면 무시하고 다음 물체 검사

                    # 사람을 찾은 경우 플래그 ON 및 좌표 추출
                    target_found = True
                    x1, y1, x2, y2 = map(int, box.xyxy[0])
                    conf = float(box.conf[0])

                    # 사람 박스의 중심점(Target) 계산
                    obj_center_x = (x1 + x2) // 2
                    obj_center_y = (y1 + y2) // 2

                    # 화면 중심점과의 픽셀 오차(Distance Error) 연산
                    error_x = obj_center_x - center_x
                    error_y = obj_center_y - center_y

                    # 시각화: 사람 락온 박스 및 중심점 그리기
                    cv2.rectangle(annotated_frame, (x1, y1), (x2, y2), (0, 255, 0), 2)
                    cv2.circle(annotated_frame, (obj_center_x, obj_center_y), 5, (0, 0, 255), -1)
                    cv2.putText(annotated_frame, f"LOCK ON PERSON ({conf:.2f})", (x1, y1 - 10),
                                cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)

                    # 여러 사람 중 첫 번째로 포착된 타겟 1명만 추적하므로 브레이크
                    break

            # -----------------------------------------------------------------
            # [새로 추가된 작업] 서보모터 제어 가중치 연산 및 아두이노 전송
            # -----------------------------------------------------------------
            if target_found and self.serial_conn and self.serial_conn.is_open:
                # P 제어 가중치 공식 (오차 비례 제어)
                # 오차 값에 이득값(0.04)을 곱해 기존 각도에서 더하거나 빼줍니다.
                # ※ 실제 구동 시 모터가 타겟의 반대 방향으로 도망치면 -= 와 += 부호를 서로 바꿔주시면 됩니다.
                self.pan_angle -= int(error_x * 0.04)
                self.tilt_angle += int(error_y * 0.04)

                # 서보모터의 물리적 파손 방지를 위한 각도 리미트 하드 가드 (0도 ~ 180도)
                self.pan_angle = max(0, min(180, self.pan_angle))
                self.tilt_angle = max(0, min(180, self.tilt_angle))

                # 아두이노 패킷 생성 (예: "P90T120\n")
                serial_cmd = f"P{self.pan_angle}T{self.tilt_angle}\n"

                # 아두이노 우노로 바이트 전송
                self.serial_conn.write(serial_cmd.encode())

                # 화면 좌상단에 실시간 제어 각도 시각화
                status_msg = f"TRACKING - Pan:{self.pan_angle} Tilt:{self.tilt_angle}"
                cv2.putText(annotated_frame, status_msg, (20, 40),
                            cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 255), 2)
            else:
                cv2.putText(annotated_frame, "SEARCHING TARGET...", (20, 40),
                            cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 0, 255), 2)

            # 화면 정중앙 타겟 조준 십자선 그리기
            cv2.line(annotated_frame, (center_x - 15, center_y), (center_x + 15, center_y), (255, 0, 0), 2)
            cv2.line(annotated_frame, (center_x, center_y - 15), (center_x, center_y + 15), (255, 0, 0), 2)

            # OpenCV 이미지를 다시 ROS 2 Image 메시지로 변환 및 발행
            result_msg = self.bridge.cv2_to_imgmsg(annotated_frame, encoding="bgr8")
            result_msg.header = msg.header
            self.publisher_.publish(result_msg)

        except Exception as e:
            self.get_logger().error(f'추론 및 제어 파이프라인 에러: {e}')


def main(args=None):
    rclpy.init(args=args)
    yolov8_node = Yolov8GimbalNode()

    try:
        rclpy.spin(yolov8_node)
    except KeyboardInterrupt:
        yolov8_node.get_logger().info('사용자에 의해 노드가 종료되었습니다.')
    finally:
        if yolov8_node.serial_conn and yolov8_node.serial_conn.is_open:
            yolov8_node.serial_conn.close()  # 종료 시 시리얼 포트 안전하게 닫기
        yolov8_node.destroy_node()
        rclpy.shutdown()

if __name__ == '__main__':
    main()

### 🛠️ 아두이노 우노(Arduino Uno) 추천 수신 스케치 코드

```
#include <Servo.h>

Servo panServo;   // 좌우 모터
Servo tiltServo;  // 상하 모터

void setup() {
  Serial.begin(115200); // 파이썬 노드와 속도 일치 필수

  panServo.attach(9);   // 좌우 서보 9번 핀
  tiltServo.attach(10); // 상하 서보 10번 핀

  // 시작 시 정중앙 정렬
  panServo.write(90);
  tiltServo.write(90);
}

void loop() {
  if (Serial.available() > 0) {
    // 줄바꿈 문자(\n)가 올 때까지 명령 문자열을 읽어옴
    String cmd = Serial.readStringUntil('\n');

    // 패킷 검증 (P로 시작하고 T가 포함되어 있는지 확인)
    int pIdx = cmd.indexOf('P');
    int tIdx = cmd.indexOf('T');

    if (pIdx != -1 && tIdx != -1) {
      // 문자열에서 팬 각도와 틸트 각도 파싱 추출
      String panStr = cmd.substring(pIdx + 1, tIdx);
      String tiltStr = cmd.substring(tIdx + 1);

      int panAngle = panStr.toInt();
      int tiltAngle = tiltStr.toInt();

      // 서보모터에 각도 명령 하달
      panServo.write(panAngle);
      tiltServo.write(tiltAngle);
    }
  }
}
